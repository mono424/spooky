## Basic Usage

<CodeBlock
  fileName="src/main.ts"
  lang="typescript"
  code={`import { SpookyClient } from '@spooky/core';

const client = new SpookyClient({
endpoint: 'ws://localhost:8000',
});

await client.connect();

// Subscribe to live query
const sub = client.liveQuery('SELECT \* FROM users').subscribe((data) => {
console.log('Updated users:', data);
});`}
